  import Foundation
  import AppIntents

  /// Donates App Intents so Siri and Shortcuts suggest them.
  /// All donation state is local; no data leaves the device.
  @available(iOS 17.0, *)
  final class IntentDonationService {
      static let shared = IntentDonationService()
      private init() {}

      private(set) var donationCount: Int = 0
      private(set) var lastDonatedAmountCents: Int64 = 0
      private(set) var lastDonatedCategory: String = ""

      /// Call this after a successful transaction save so Siri learns the pattern.
      func donateExpenseLogged(amountCents: Int64, categoryName: String) {
          lastDonatedAmountCents = amountCents
          lastDonatedCategory = categoryName
          donationCount += 1

          // Donate to the system so Shortcuts suggests it
          let intent = LogQuickExpenseIntent()
          intent.amount = Double(amountCents) / 100.0
          intent.categoryName = categoryName
          Task {
              try? await intent.donate(result: nil)
          }
      }

      /// Donate the "check allowance" intent after the user views the dashboard.
      /// Keeps Siri suggestions relevant.
      func donateAllowanceChecked() {
          let intent = CheckAllowanceIntent()
          Task {
              try? await intent.donate(result: nil)
          }
      }

      #if DEBUG
      func resetForTesting() {
          donationCount = 0
          lastDonatedAmountCents = 0
          lastDonatedCategory = ""
      }
      #endif
  }
  ```

  @available(iOS 17.0, *)
  struct LogQuickExpenseIntent: AppIntent {
      static var title: LocalizedStringResource = "Log Quick Expense"
      static var description = IntentDescription(
          "Logs an expense with a preset amount. Used by interactive widgets and shortcuts."
      )

      @Parameter(title: "Amount")
      var amount: Double?

      @Parameter(title: "Category")
      var categoryName: String?

      init() {}

      init(amount: Double, categoryName: String) {
          self.amount = amount
          self.categoryName = categoryName
      }

      func perform() async throws -> some IntentResult {
          // Post notification for the main app to handle (same pattern as AddExpenseIntent)
          var userInfo: [AnyHashable: Any] = [:]
          if let amount { userInfo["amount"] = amount }
          if let categoryName { userInfo["category"] = categoryName }
          userInfo["source"] = "quickLog"
          NotificationCenter.default.post(
              name: .openTransactionEntry,
              object: nil,
              userInfo: userInfo
          )
          return .result()
      }
  }

  @available(iOS 17.0, *)
  struct CheckAllowanceIntent: AppIntent {
      static var title: LocalizedStringResource = "Check Daily Allowance"
      static var description = IntentDescription(
          "Reads your remaining daily allowance from BudgetVault."
      )

      init() {}

      func perform() async throws -> some IntentResult & ProvidesDialog {
          let data = WidgetDataService.readCurrentData()
          let formatted = CurrencyFormatter.format(
              cents: data.dailyAllowanceCents,
              currencyCode: data.currencyCode
          )
          let dialog = IntentDialog("You have \(formatted) to spend today.")
          return .result(dialog: dialog)
      }
  }
  ```

  ```swift
  import Foundation

  /// After onboarding, posts a deep-link notification that DashboardView
  /// handles to open TransactionEntryView pre-filled with a gentle example.
  /// Only fires once per device via UserDefaults flag.
  enum OnboardingFirstExpenseService {
      private static let didSeedKey = "v33_firstExpenseSeeded"

      static var shouldSeed: Bool {
          !UserDefaults.standard.bool(forKey: didSeedKey)
      }

      static func seedFirstExpense() {
          guard shouldSeed else { return }
          UserDefaults.standard.set(true, forKey: didSeedKey)

          // Delay so onboarding dismiss animation completes first
          DispatchQueue.main.asyncAfter(deadline: .now() + 0.9) {
              NotificationCenter.default.post(
                  name: .openTransactionEntry,
                  object: nil,
                  userInfo: [
                      "amount": 4.50,
                      "category": "Food",
                      "note": "Coffee",
                      "source": "onboardingSeed"
                  ]
              )
          }
      }

      #if DEBUG
      static func resetForTesting() {
          UserDefaults.standard.removeObject(forKey: didSeedKey)
      }
      #endif
  }
  ```

  ```swift
  import Foundation

  /// Gates celebration-context paywall presentations so they fire at most
  /// once per month per trigger, and never for premium users.
  enum CelebrationTriggerService {
      enum Trigger: String {
          case postWrapped
          case postAchievement
          case postMonthEndUnderBudget
      }

      private static func lastShownKey(for trigger: Trigger) -> String {
          "celebrationPaywall_\(trigger.rawValue)_lastShown"
      }

      /// Returns true iff the paywall should be shown for this trigger right now.
      /// Gating rules:
      ///  - Never for premium users.
      ///  - Max once per calendar month per trigger.
      static func shouldShow(trigger: Trigger, isPremium: Bool) -> Bool {
          guard !isPremium else { return false }
          let lastShown = UserDefaults.standard.object(
              forKey: lastShownKey(for: trigger)
          ) as? Date
          guard let lastShown else { return true }
          return !Calendar.current.isDate(
              lastShown, equalTo: Date(), toGranularity: .month
          )
      }

      /// Records that a celebration paywall was just shown.
      static func markShown(_ trigger: Trigger) {
          UserDefaults.standard.set(
              Date(), forKey: lastShownKey(for: trigger)
          )
      }

      #if DEBUG
      static func resetAllForTesting() {
          for t in [Trigger.postWrapped, .postAchievement, .postMonthEndUnderBudget] {
              UserDefaults.standard.removeObject(forKey: lastShownKey(for: t))
          }
      }
      #endif
  }
  ```

## Decisions log

- **Kept friction-based paywall triggers alongside celebration triggers.** Don't remove what's converting; just add better-converting triggers on top.
- **Gated celebration triggers to once-per-month-per-trigger.** A user who sees three paywalls in one Wrapped session is a worse experience than a missed conversion.
- **`streak_14` is Silver tier (not Bronze).** It's a real commitment; tier should reflect it.
- **First-expense seed uses $4.50 Coffee / Food.** Universal, non-judgmental, matches the "one-tap demo" pattern from successful apps like Duolingo.
- **Donation service writes state synchronously but donates async.** Test determinism comes first; Siri learning can lag a frame.
