```bash
#!/usr/bin/env bash
set -euo pipefail

# Compute build number from total commit count on HEAD. Monotonic, never collides.
BUILD=$(git rev-list --count HEAD)

# Update both targets in project.yml in place using yq if available, else sed.
PROJECT_YML="$(dirname "$0")/../project.yml"

if command -v yq >/dev/null 2>&1; then
  yq -i ".targets.BudgetVault.settings.base.CURRENT_PROJECT_VERSION = \"${BUILD}\"" "$PROJECT_YML"
  yq -i ".targets.BudgetVaultWidgetExtension.settings.base.CURRENT_PROJECT_VERSION = \"${BUILD}\"" "$PROJECT_YML"
else
  /usr/bin/sed -i '' "s/CURRENT_PROJECT_VERSION: \".*\"/CURRENT_PROJECT_VERSION: \"${BUILD}\"/g" "$PROJECT_YML"
fi

# Generate release notes from commit subjects since last tag.
LAST_TAG=$(git describe --tags --abbrev=0 2>/dev/null || echo "")
NOTES_DIR="$(dirname "$0")/../fastlane/metadata/en-US"
mkdir -p "$NOTES_DIR"

if [ -n "$LAST_TAG" ]; then
  git log "${LAST_TAG}..HEAD" --pretty=format:"- %s" --no-merges > "$NOTES_DIR/release_notes.txt"
else
  git log --pretty=format:"- %s" --no-merges -n 20 > "$NOTES_DIR/release_notes.txt"
fi

echo "Build: ${BUILD}"
echo "Release notes written to ${NOTES_DIR}/release_notes.txt"
```

curl -O -L "https://github.com/actions/runner/releases/download/v${RUNNER_VERSION}/actions-runner-${RUNNER_ARCH}-${RUNNER_VERSION}.tar.gz"

  --token "${TOKEN}" \
  --name "$(hostname -s)-runner" \
  --labels "self-hosted,macOS,ARM64" \
  --work "_work" \
  --unattended \
  --replace

# Install as a launchd service so it auto-starts on login.
./svc.sh install
./svc.sh start

echo "Runner installed and started. Verify at https://github.com/${REPO}/settings/actions/runners"

```swift
import Foundation
import MetricKit
import os

/// On-device crash + hang + disk telemetry via Apple's MetricKit.
/// Privacy-safe: payloads never leave the device unless the user explicitly
/// attaches them to a Send Feedback email.
///
/// Replaces Sentry/Crashlytics/Bugsnag. App Store privacy label remains
/// "Data Not Collected" because no network call is ever made.
@MainActor
final class MetricService: NSObject, MXMetricManagerSubscriber {

    static let shared = MetricService()
    private let logger = Logger(subsystem: "io.budgetvault.app", category: "MetricKit")

    private override init() { super.init() }

    /// Call once from `BudgetVaultApp.init()`. Idempotent.
    static func start() {
        MXMetricManager.shared.add(shared)
        try? FileManager.default.createDirectory(
            at: diagnosticsDirectory(),
            withIntermediateDirectories: true
        )
        excludeFromBackup(diagnosticsDirectory())
    }

    // MARK: MXMetricManagerSubscriber

    nonisolated func didReceive(_ payloads: [MXMetricPayload]) {
        for payload in payloads {
            persist(data: payload.jsonRepresentation(),
                    kind: "metric",
                    timestamp: payload.timeStampEnd)
        }
    }

    nonisolated func didReceive(_ payloads: [MXDiagnosticPayload]) {
        for payload in payloads {
            persist(data: payload.jsonRepresentation(),
                    kind: "diagnostic",
                    timestamp: payload.timeStampEnd)
        }
    }

    // MARK: Persistence

    nonisolated private func persist(data: Data, kind: String, timestamp: Date) {
        let dir = Self.diagnosticsDirectory()
        let formatter = ISO8601DateFormatter()
        let filename = "\(kind)-\(formatter.string(from: timestamp)).json"
        let url = dir.appendingPathComponent(filename)
        try? data.write(to: url, options: [.atomic, .completeFileProtection])
        logger.info("Persisted \(kind) payload at \(url.lastPathComponent)")
    }

    static func diagnosticsDirectory() -> URL {
        let docs = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask)[0]
        return docs.appendingPathComponent("diagnostics", isDirectory: true)
    }

    static func excludeFromBackup(_ url: URL) {
        var values = URLResourceValues()
        values.isExcludedFromBackup = true
        var mutableURL = url
        try? mutableURL.setResourceValues(values)
    }

    // MARK: Public read API for FeedbackService

    /// Returns up to `limit` most recent diagnostic payloads as Data, newest first.
    static func recentPayloads(limit: Int = 5) -> [(name: String, data: Data)] {
        let dir = diagnosticsDirectory()
        guard let urls = try? FileManager.default.contentsOfDirectory(
            at: dir,
            includingPropertiesForKeys: [.contentModificationDateKey]
        ) else { return [] }

        let sorted = urls.sorted { lhs, rhs in
            let l = (try? lhs.resourceValues(forKeys: [.contentModificationDateKey]).contentModificationDate) ?? .distantPast
            let r = (try? rhs.resourceValues(forKeys: [.contentModificationDateKey]).contentModificationDate) ?? .distantPast
            return l > r
        }
        return sorted.prefix(limit).compactMap { url in
            guard let data = try? Data(contentsOf: url) else { return nil }
            return (url.lastPathComponent, data)
        }
    }
}
```

        try Data("new".utf8).write(to: newURL)

        let result = MetricService.recentPayloads(limit: 2)
        XCTAssertEqual(result.first?.name, "metric-new.json")
    }
}

import SwiftUI

/// Vault Motion Language — a single vocabulary of motion fired on reward
/// states throughout the app. Reuses `VaultDialMark` and `ConfettiView`
/// (no new visual primitives). See docs/BRAND.md §4.
///
/// Usage:
///   .vaultCelebration(trigger: $celebration)
/// where `celebration` is `VaultCelebration.Kind?`. Setting it to a
/// non-nil case fires that motion; the modifier resets it to nil
/// after the animation completes.
enum VaultCelebration {

    enum Kind: Equatable {
        /// Month close, no-spend day tap, reconciled week. Brief tumbler-snap.
        case sealed
        /// Milestone unlock, premium purchase. Full dial spin with combo readout.
        case dialSpin(combo: String?)
        /// First reconcile, small wins. One soft tumbler click.
        case tumblerClick
    }
}

extension View {
    /// Attach the Vault Motion Language modifier to any view. The bound
    /// trigger fires the motion exactly once and then resets to nil.
    func vaultCelebration(trigger: Binding<VaultCelebration.Kind?>) -> some View {
        modifier(VaultCelebrationModifier(trigger: trigger))
    }
}

private struct VaultCelebrationModifier: ViewModifier {
    @Binding var trigger: VaultCelebration.Kind?
    @Environment(\.accessibilityReduceMotion) private var reduceMotion

    @State private var dialRotation: Double = 0
    @State private var dialScale: CGFloat = 0.6
    @State private var dialOpacity: Double = 0
    @State private var captionOpacity: Double = 0
    @State private var captionText: String = ""
    @State private var showConfetti: Bool = false

    func body(content: Content) -> some View {
        ZStack {
            content
            overlay
                .allowsHitTesting(false)
        }
        .onChange(of: trigger) { _, new in
            guard let kind = new else { return }
            fire(kind)
        }
    }

    @ViewBuilder
    private var overlay: some View {
        if dialOpacity > 0.01 {
            VStack(spacing: BudgetVaultTheme.spacingMD) {
                VaultDialMark(size: 96, color: .white, showGlow: true, tickRotation: dialRotation)
                    .scaleEffect(dialScale)
                    .opacity(dialOpacity)
                if !captionText.isEmpty {
                    Text(captionText)
                        .font(.headline)
                        .foregroundStyle(.white)
                        .opacity(captionOpacity)
                        .accessibilityAddTraits(.isStaticText)
                }
            }
            .frame(maxWidth: .infinity, maxHeight: .infinity)
            .background(.black.opacity(0.35).ignoresSafeArea())
            .accessibilityElement(children: .combine)
            .accessibilityAddTraits(.updatesFrequently)
        }
        if showConfetti {
            ConfettiView(isActive: showConfetti, style: .coins, particleCount: 28, duration: 2.4)
        }
    }

    private func fire(_ kind: VaultCelebration.Kind) {
        let duration: Double = reduceMotion ? 0.4 : 1.6

        switch kind {
        case .sealed:
            captionText = "Vault sealed."
            HapticManager.notification(.success)
            animateSeal(duration: duration)

        case .dialSpin(let combo):
            captionText = combo ?? "Combination accepted."
            HapticManager.notification(.success)
            showConfetti = !reduceMotion
            animateSpin(duration: duration)

        case .tumblerClick:
            captionText = "Tumbler clicks into place."
            HapticManager.impact(.light)
            animateClick(duration: duration)
        }

        DispatchQueue.main.asyncAfter(deadline: .now() + duration + 0.4) {
            trigger = nil
            showConfetti = false
        }
    }

    private func animateSeal(duration: Double) {
        withAnimation(.spring(response: 0.4, dampingFraction: 0.8)) {
            dialOpacity = 1
            dialScale = 1
            captionOpacity = 1
        }
        withAnimation(.easeOut(duration: 0.5)) {
            dialRotation = reduceMotion ? 0 : -22
        }
        withAnimation(.easeIn(duration: 0.4).delay(duration - 0.4)) {
            dialOpacity = 0
            captionOpacity = 0
        }
    }

    private func animateSpin(duration: Double) {
        withAnimation(.spring(response: 0.5, dampingFraction: 0.7)) {
            dialOpacity = 1
            dialScale = 1
            captionOpacity = 1
        }
        withAnimation(.easeInOut(duration: duration).delay(0.05)) {
            dialRotation = reduceMotion ? 0 : 1080
        }
        withAnimation(.easeIn(duration: 0.4).delay(duration - 0.3)) {
            dialOpacity = 0
            captionOpacity = 0
        }
    }

    private func animateClick(duration: Double) {
        withAnimation(.spring(response: 0.3, dampingFraction: 0.6)) {
            dialOpacity = 1
            dialScale = 1
            captionOpacity = 1
        }
        withAnimation(.easeOut(duration: 0.25)) {
            dialRotation = reduceMotion ? 0 : -12
        }
        withAnimation(.easeIn(duration: 0.3).delay(duration - 0.3)) {
            dialOpacity = 0
            captionOpacity = 0
        }
    }
}

        /// Tumbler-snap — used for terminal "done" states (no follow-up action).
        case tumblerSnap
    }

    let icon: String
    let title: String
    let message: String
    var actionLabel: String? = nil
    var action: (() -> Void)? = nil
    var motion: Motion = .doorSwing

    @Environment(\.accessibilityReduceMotion) private var reduceMotion
    @State private var animPhase: CGFloat = 0

    var body: some View {
        VStack(spacing: BudgetVaultTheme.spacingMD) {
            Image(systemName: icon)
                .font(BudgetVaultTheme.iconLarge)
                .foregroundStyle(.secondary)
                .rotationEffect(.degrees(rotationDegrees))
                .scaleEffect(scaleAmount)
                .onAppear { startMotion() }
            Text(title)
                .font(.title3.bold())
            Text(message)
                .font(.body)
                .foregroundStyle(.secondary)
                .multilineTextAlignment(.center)

            if let actionLabel, let action {
                Button(actionLabel, action: action)
                    .buttonStyle(PrimaryButtonStyle())
                    .padding(.top, 4)
            }
        }
        .padding(BudgetVaultTheme.spacingPage)
        .frame(maxWidth: .infinity, maxHeight: .infinity)
        .accessibilityElement(children: .combine)
    }

    private var rotationDegrees: Double {
        guard !reduceMotion else { return 0 }
        switch motion {
        case .doorSwing:   return Double(animPhase) * -8
        case .dialSpin:    return Double(animPhase) * 360
        case .tumblerSnap: return 0
        }
    }

    private var scaleAmount: CGFloat {
        guard !reduceMotion else { return 1 }
        switch motion {
        case .doorSwing:   return 1
        case .dialSpin:    return 1
        case .tumblerSnap: return 1 + animPhase * 0.06
        }
    }

    private func startMotion() {
        guard !reduceMotion else { return }
        switch motion {
        case .doorSwing:
            withAnimation(.easeInOut(duration: 2.4).repeatForever(autoreverses: true)) {
                animPhase = 1
            }
        case .dialSpin:
            withAnimation(.linear(duration: 4.0).repeatForever(autoreverses: false)) {
                animPhase = 1
            }
        case .tumblerSnap:
            withAnimation(.spring(response: 0.4, dampingFraction: 0.5).repeatForever(autoreverses: true)) {
                animPhase = 1
            }
        }
    }
}

# v3.4 CKShare Schema Design

**Status:** Design only. Implementation lands in v3.4.
**Author:** v3.3.1 planning pass
**Date:** 2026-04-16

## Goal
Enable two-person household sharing on top of the existing SwiftData + CloudKit
private database, without breaking the v3.0–v3.3 single-user model.

## Non-goals
- 3+ person sharing (deferred indefinitely)
- Sharing arbitrary historical budgets (only the active period + future is shared)
- Cross-account merge / unmerge (v3.5 problem)

## Architecture: sharing-root model

CKShare in CloudKit requires a single "root record" per share. We introduce a
`SharedHousehold` SwiftData model that owns its own zone:

```swift
@Model
final class SharedHousehold {
    @Attribute(.unique) var id: UUID
    var displayName: String
    var ownerUserRecordName: String
    var createdAt: Date
    @Relationship(deleteRule: .cascade) var sharedBudgets: [Budget]?
}
```

Existing `Budget` gets an optional back-reference:

```swift
@Model
final class Budget {
    // ... existing fields ...
    var household: SharedHousehold?  // nil = personal vault, non-nil = shared
}
```

## Zone partitioning

| Zone | Contents | CKShare? |
|---|---|---|
| `_defaultZone` (private) | All current personal data — Budgets with `household == nil`, Categories, Transactions, RecurringExpenses, all v3.3 entities | No |
| Custom zone `household-<UUID>` | One per `SharedHousehold` plus its `Budget` graph | Yes (one CKShare per household) |

A user may belong to at most one household at a time in v3.4. Personal and
shared budgets coexist in the UI under separate tabs.

## Migration plan: V2 → V3

V3 adds:
- `SharedHousehold` model (new entity)
- `Budget.household` optional relationship (additive, default nil)
- No changes to `Category`, `Transaction`, `RecurringExpense`

Migration is therefore lightweight (no data transform, no field renames).
Existing budgets default `household = nil` and remain personal.

## Sharing flow (UX summary, for reference only)

1. Owner taps "Invite partner" in Settings → Vault.
2. App calls `CKContainer.share(SharedHousehold)` → returns CKShare URL.
3. Owner sends URL via system share sheet.
4. Partner taps URL on their device → CloudKit accept flow → SwiftData picks up
   the new shared zone via `NSPersistentCloudKitContainer.acceptShareInvitations`.
5. Both users now read+write the same budget. Last-write-wins via CloudKit's
   default conflict resolution; cents-as-Int64 means no Decimal corruption.

## Open questions for v3.4 implementation

1. Permission model: read-only vs. read-write (recommend read-write only for v3.4).
2. Categories: should they be shared (same envelopes) or per-user (each partner
   sees their own categorisation)? Recommend shared.
3. Transaction author display: add `Transaction.authorRecordName` so the UI can
   show "Logged by Alex"? Yes — additive in V3.
4. Removal flow: when a share is revoked, do shared budgets convert to personal
   on the leaving user's device? Recommend: yes, owned by the recipient as
   read-only archive.

## What this design preserves

- Cascade rules: `SharedHousehold → Budget → Category → Transaction` honored end-to-end.
- No `#Unique` macro (per architecture rule 6).
- VersionedSchema pattern (V1 → V2 → V3).
- CloudKit-only sharing; no third-party servers, no Plaid, no analytics SDK.
- Privacy label remains "Data Not Collected" (CKShare uses Apple infrastructure).

## Hard constraint

This document does NOT ship code in v3.3.1. It is a design contract for v3.4.
v3.3.1 ships `BudgetVaultSchemaV2` (indexes only) — V3 adding `SharedHousehold`
ships in v3.4.
